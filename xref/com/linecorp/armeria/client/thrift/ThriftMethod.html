39  @SuppressWarnings("rawtypes")
40  class ThriftMethod {
41      private final boolean oneWay;
42      private final String name;
43      final Class<?>[] declaredThrowableException;
44      private final TBase<? extends TBase, TFieldIdEnum> argsObject;
45      private final TFieldIdEnum[] argsFieldIdEnums;
46  
47      private final TBase<? extends TBase, TFieldIdEnum> resultObject;
48      private final TFieldIdEnum successField;
49      private final List<TFieldIdEnum> exceptionFields;
50  
51      private final List<Class<?>> paramTypes;
52      private final Class<?> returnType;
53  
54      @SuppressWarnings({ "unchecked", "SuspiciousArrayCast" })
55      ThriftMethod(Class<?> clientClass, Method method, String thriftServiceName) {
56          requireNonNull(clientClass);
57          requireNonNull(method);
58          requireNonNull(thriftServiceName);
59          name = method.getName();
60          declaredThrowableException = method.getExceptionTypes();
61  
62          boolean oneWay = false;
63          try {
64              clientClass.getMethod("recv_" + name);
65          } catch (NoSuchMethodException ignore) {
66              oneWay = true;
67          }
68          this.oneWay = oneWay;
69  
70          String argClassName = thriftServiceName + '$' + name + "_args";
71          final Class<TBase<? extends TBase, TFieldIdEnum>> argClass;
72          try {
73              argClass = (Class<TBase<? extends TBase, TFieldIdEnum>>) Class.forName(argClassName);
74              argsObject = argClass.newInstance();
75          } catch (Exception e) {
76              throw new IllegalArgumentException("fail to create a new instance: " + argClassName, e);
77          }
78  
79          String argFieldEnumName = thriftServiceName + '$' + name + "_args$_Fields";
80          try {
81              Class<?> fieldIdEnumClass = Class.forName(argFieldEnumName);
82              argsFieldIdEnums = (TFieldIdEnum[]) requireNonNull(fieldIdEnumClass.getEnumConstants(),
83                                                                 "field enum may not be empty");
84          } catch (Exception e) {
85              throw new IllegalArgumentException("fail to create a new instance : " + argFieldEnumName, e);
86          }
87  
88          FieldValueMetaData successFieldMetadata = null;
89          if (oneWay) {
90              resultObject = null;
91              successField = null;
92              exceptionFields = Collections.emptyList();
93          } else {
94              String resultClassName = thriftServiceName + '$' + name + "_result";
95              final Class resultClass;
96              try {
97                  resultClass = Class.forName(resultClassName);
98                  resultObject = (TBase<? extends TBase, TFieldIdEnum>) resultClass.newInstance();
99              } catch (Exception e) {
100                 throw new IllegalArgumentException("fail to create a new instance : " + resultClassName, e);
101             }
102 
103             try {
104 
105                 @SuppressWarnings("unchecked")
106                 final Map<TFieldIdEnum, FieldMetaData> resultMetaDataMap =
107                         (Map<TFieldIdEnum, FieldMetaData>) FieldMetaData.getStructMetaDataMap(resultClass);
108 
109                 TFieldIdEnum successField = null;
110                 List<TFieldIdEnum> exceptionFields = new ArrayList<>(resultMetaDataMap.size());
111                 for (Entry<TFieldIdEnum, FieldMetaData> e : resultMetaDataMap.entrySet()) {
112                     final TFieldIdEnum key = e.getKey();
113                     final String fieldName = key.getFieldName();
114                     if ("success".equals(fieldName)) {
115                         successField = key;
116                         successFieldMetadata = e.getValue().valueMetaData;
117                         continue;
118                     }
119 
120                     Class<?> fieldType = resultClass.getField(fieldName).getType();
121                     if (Throwable.class.isAssignableFrom(fieldType)) {
122                         exceptionFields.add(key);
123                     }
124                 }
125 
126                 this.successField = successField;
127                 this.exceptionFields = Collections.unmodifiableList(exceptionFields);
128             } catch (Exception e) {
129                 throw new IllegalArgumentException(
130                         "failed to find the result metaDataMap: " + resultClass.getName(),
131                         e);
132             }
133         }
134 
135         // Determine the parameter types of the function.
136         paramTypes = Collections.unmodifiableList(
137                 FieldMetaData.getStructMetaDataMap(argClass).values().stream()
138                              .map(e -> ThriftUtil.toJavaType(e.valueMetaData)).collect(Collectors.toList()));
139 
140         if (successFieldMetadata != null) {
141             returnType = ThriftUtil.toJavaType(successFieldMetadata);
142         } else {
143             returnType = Void.class;
144         }
145     }
146 
147     TBase createArgs() {
148         return argsObject.deepCopy();
149     }
150 
151     @SuppressWarnings("unchecked")
152     TBase createArgs(boolean isAsync, Object[] args) {
153         final TBase newArgs = createArgs();
154         if (args != null) {
155             final int toFillArgLength = args.length - (isAsync ? 1 : 0);
156             for (int i = 0; i < toFillArgLength; i++) {
157                 newArgs.setFieldValue(argsFieldIdEnums[i], args[i]);
158             }
159         }
160         return newArgs;
161     }
162 
163     boolean isOneWay() {
164         return oneWay;
165     }
166 
167     byte methodType() {
168         return oneWay ? TMessageType.ONEWAY : TMessageType.CALL;
169     }
170 
171     List<Class<?>> paramTypes() {
172         return paramTypes;
173     }
174 
175     Class<?> returnType() {
176         return returnType;
177     }
178 
179     static AsyncMethodCallback asyncCallback(Object[] args) {
180         if (requireNonNull(args, "args").length == 0) {
181             throw new IllegalArgumentException("args must contains objects");
182         }
183         final Object lastObj = args[args.length - 1];
184         if (lastObj instanceof AsyncMethodCallback) {
185             return (AsyncMethodCallback) lastObj;
186         }
187         if (lastObj == null) {
188             return null;
189         }
190         throw new IllegalArgumentException(
191                 "the last element of args must be AsyncMethodCallback: " + lastObj.getClass().getName());
192     }
193 
194     static AsyncMethodCallback asyncMethodCallbackFromContext(ServiceInvocationContext ctx) {
195         requireNonNull(ctx, "ctx");
196         List<?> params = ctx.params();
197         if (params == null || params.isEmpty()) {
198             return null;
199         }
200         Object lastObj = params.get(params.size() - 1);
201         return lastObj instanceof AsyncMethodCallback ? (AsyncMethodCallback) lastObj : null;
202     }
203 
204     Class<?>[] declaredThrowableException() {
205         return declaredThrowableException;
206     }
207 
208     TBase<? extends TBase, TFieldIdEnum> createResult() {
209         return resultObject.deepCopy();
210     }
211 
212     TFieldIdEnum successField() {
213         return successField;
214     }
215 
216     List<TFieldIdEnum> getExceptionFields() {
217         return exceptionFields;
218     }
