63      /**
64       * 2^29 - We could have used 2^30 but this should be large enough.
65       */
66      private static final int MAX_NUM_REQUESTS_SENT = 536870912;
67  
68      private static final String ARMERIA_USER_AGENT = "armeria client";
69  
70      static HttpSession get(Channel ch) {
71          final HttpSessionHandler sessionHandler = ch.pipeline().get(HttpSessionHandler.class);
72          if (sessionHandler == null) {
73              return HttpSession.INACTIVE;
74          }
75  
76          return sessionHandler;
77      }

79      private final HttpSessionChannelFactory channelFactory;
80      private final Promise<Channel> sessionPromise;
81      private final ScheduledFuture<?> timeoutFuture;

83      /** Whether the current channel is active or not **/
84      private volatile boolean active;

86      /** The current negotiated {@link SessionProtocol} */
87      private SessionProtocol protocol;

89      /**
90       * A queue of pending requests.
91       * Set to {@link SequentialWaitsHolder} or {@link MultiplexWaitsHolder} later on protocol negotiation.

93      private WaitsHolder waitsHolder = PreNegotiationWaitsHolder.INSTANCE;

95      /** The number of requests sent. Disconnects when it reaches at {@link #MAX_NUM_REQUESTS_SENT}. */
96      private int numRequestsSent;

99       * {@code true} if the protocol upgrade to HTTP/2 has failed.
100      * If set to {@code true}, another connection attempt will follow.
101      */
102     private boolean needsRetryWithH1C;

106

132     private boolean isDisconnectionRequired(FullHttpResponse response) {
133         if (isDisconnectionPending(numRequestsSent) && waitsHolder.isEmpty()) {
134             return true;
135         }
136 
137         // HTTP/1 request with the 'Connection: close' header
138         return !protocol().isMultiplex() &&
139                HttpHeaderValues.CLOSE.contentEqualsIgnoreCase(
140                        response.headers().get(HttpHeaderNames.CONNECTION));
141     }
142 
143     private static boolean isDisconnectionPending(int numRequestsSent) {
144         return numRequestsSent >= MAX_NUM_REQUESTS_SENT;
145     }

147     @Override
148     public void deactivate() {
149         active = false;

163     @Override
164     public void channelRead(ChannelHandlerContext ctx, Object msg) throws Exception {
165         if (msg instanceof Http2Settings) {
166             // Expected
167         } else if (msg instanceof FullHttpResponse) {
168             FullHttpResponse response = (FullHttpResponse) msg;
169 
170             final Invocation invocation = waitsHolder.poll(response);
171 
172             if (invocation != null) {
173                 final ServiceInvocationContext iCtx = invocation.invocationContext();
174                 final SerializationFormat serializationFormat = iCtx.scheme().serializationFormat();
175                 try {
176                     final Promise<FullHttpResponse> resultPromise = invocation.resultPromise();
177                     if (HttpStatusClass.SUCCESS == response.status().codeClass()
178                         // No serialization indicates a raw HTTP protocol which should
179                         // have error responses returned.
180                         || serializationFormat == SerializationFormat.NONE) {
181                         iCtx.resolvePromise(resultPromise, response.retain());
182                     } else {
183                         final DecoderResult decoderResult = response.decoderResult();
184                         final Throwable cause;
185 
186                         if (decoderResult.isSuccess()) {
187                             cause = new InvalidResponseException("HTTP Response code: " + response.status());
188                         } else {
189                             final Throwable decoderCause = decoderResult.cause();
190                             if (decoderCause instanceof DecoderException) {
191                                 cause = decoderCause;
192                             } else {
193                                 cause = new DecoderException("protocol violation: " + decoderCause,
194                                                              decoderCause);
195                             }
196                         }
197 
198                         iCtx.rejectPromise(resultPromise, cause);
199                     }
200                 } finally {
201                     ReferenceCountUtil.release(msg);
202                 }
203             } else {
204                 logger.warn("{} Received a response without a matching request: {}", ctx.channel(), msg);
205             }
206 
207             if (isDisconnectionRequired(response)) {
208                 ctx.close();
209             }
210         } else {
211             try {
212                 final String typeInfo;
213                 if (msg instanceof ByteBuf) {
214                     typeInfo = msg + " HexDump: " + ByteBufUtil.hexDump((ByteBuf) msg);
215                 } else {
216                     typeInfo = String.valueOf(msg);
217                 }
218                 throw new IllegalStateException("unexpected message type: " + typeInfo);
219             } finally {
220                 ReferenceCountUtil.release(msg);
221             }
222         }
223     }

225     @Override
226     public void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception {
227         if (evt instanceof SessionProtocol) {
228             assert protocol == null;
229             assert waitsHolder instanceof PreNegotiationWaitsHolder;
230 
231             timeoutFuture.cancel(false);
232 
233             // Set the current protocol and its associated WaitsHolder implementation.
234             final SessionProtocol protocol = (SessionProtocol) evt;
235             this.protocol = protocol;
236             waitsHolder = protocol.isMultiplex() ? new MultiplexWaitsHolder() : new SequentialWaitsHolder();
237 
238             if (!sessionPromise.trySuccess(ctx.channel())) {
239                 // Session creation has been failed already; close the connection.
240                 ctx.close();
241             }
242             return;
243         }
244 
245         if (evt instanceof SessionProtocolNegotiationException) {
246             timeoutFuture.cancel(false);
247             sessionPromise.tryFailure((SessionProtocolNegotiationException) evt);
248             ctx.close();
249             return;
250         }
251 
252         logger.warn("{} Unexpected user event: {}", ctx.channel(), evt);
253     }
254 
255     @Override
256     public void channelInactive(ChannelHandlerContext ctx) throws Exception {
257         active = false;
258 
259         // Protocol upgrade has failed, but needs to retry.
260         if (needsRetryWithH1C) {
261             assert waitsHolder.isEmpty();
262             timeoutFuture.cancel(false);
263             channelFactory.connect(ctx.channel().remoteAddress(), SessionProtocol.H1C, sessionPromise);
264         } else {
265             // Fail all pending responses.
266             failPendingResponses(ClosedSessionException.INSTANCE);
267 
268             // Cancel the timeout and reject the sessionPromise just in case the connection has been closed
269             // even before the session protocol negotiation is done.
270             timeoutFuture.cancel(false);
271             sessionPromise.tryFailure(ClosedSessionException.INSTANCE);
272         }
273     }

276     @Override
277     public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
278         Exceptions.logIfUnexpected(logger, ctx.channel(), protocol(), cause);
279         if (ctx.channel().isActive()) {
280             ctx.close();
281         }
282     }
283 
284     private void failPendingResponses(Throwable e) {
285         final Collection<Invocation> invocations = waitsHolder.getAll();
286         if (!invocations.isEmpty()) {
287             invocations.forEach(i -> i.invocationContext().rejectPromise(i.resultPromise(), e));
288             waitsHolder.clear();
289         }
290     }
291 
292     @Override
293     public void write(ChannelHandlerContext ctx, Object msg, ChannelPromise promise) throws Exception {
294         if (msg instanceof Invocation) {
295             Invocation invocation = (Invocation) msg;
296             FullHttpRequest request = convertToHttpRequest(invocation);
297             waitsHolder.put(invocation, request);
298             ctx.write(request, promise);
299         } else {
300             ctx.write(msg, promise);
301         }
302     }

351     private static final class SequentialWaitsHolder implements WaitsHolder {
352 
353         private final Queue<Invocation> requestExpectQueue;
354 
355         SequentialWaitsHolder() {
356             requestExpectQueue = new ArrayDeque<>();
357         }
358 
359         @Override
360         public Invocation poll(FullHttpResponse response) {
361             return requestExpectQueue.poll();
362         }
363 
364         @Override
365         public void put(Invocation invocation, FullHttpRequest request) {
366             requestExpectQueue.add(invocation);
367         }
368 
369         @Override
370         public Collection<Invocation> getAll() {
371             return requestExpectQueue;
372         }
373 
374         @Override
375         public void clear() {
376             requestExpectQueue.clear();
377         }
378 
379         @Override
380         public int size() {
381             return requestExpectQueue.size();
382         }
383 
384         @Override
385         public boolean isEmpty() {
386             return requestExpectQueue.isEmpty();
387         }
388     }

390     private static final class MultiplexWaitsHolder implements WaitsHolder {
391 
392         private final IntObjectMap<Invocation> resultExpectMap;
393         private int streamId;
394 
395         MultiplexWaitsHolder() {
396             resultExpectMap = new IntObjectHashMap<>();
397             streamId = 1;
398         }
399 
400         @Override
401         public Invocation poll(FullHttpResponse response) {
402             int streamID = response.headers().getInt(ExtensionHeaderNames.STREAM_ID.text(), 0);
403             return resultExpectMap.remove(streamID);
404         }
405 
406         @Override
407         public void put(Invocation invocation, FullHttpRequest request) {
408             int streamId = nextStreamID();
409             request.headers().add(ExtensionHeaderNames.STREAM_ID.text(), streamIdToString(streamId));
410             resultExpectMap.put(streamId, invocation);
411         }
412 
413         @Override
414         public Collection<Invocation> getAll() {
415             return resultExpectMap.values();
416         }
417 
418         @Override
419         public void clear() {
420             resultExpectMap.clear();
421         }
422 
423         @Override
424         public int size() {
425             return resultExpectMap.size();
426         }
427 
428         @Override
429         public boolean isEmpty() {
430             return resultExpectMap.isEmpty();
431         }
432 
433         private static String streamIdToString(int streamID) {
434             return Integer.toString(streamID);
435         }
436 
437         private int nextStreamID() {
438             return streamId += 2;
439         }
440     }

442     private FullHttpRequest convertToHttpRequest(Invocation invocation) {
443         requireNonNull(invocation, "invocation");
444         final ServiceInvocationContext ctx = invocation.invocationContext();
445         final FullHttpRequest request;
446         final Object content = invocation.content();
447 
448         if (content instanceof FullHttpRequest) {
449             request = (FullHttpRequest) content;
450         } else if (content instanceof ByteBuf) {
451             request = new DefaultFullHttpRequest(HttpVersion.HTTP_1_1, HttpMethod.POST, ctx.path(),
452                                                  (ByteBuf) content);
453         } else {
454             throw new IllegalStateException(
455                     "content is not a ByteBuf or FullHttpRequest: " + content.getClass().getName());
456         }
457 
458         HttpHeaders headers = request.headers();
459 
460         headers.set(HttpHeaderNames.HOST, hostHeader(ctx));
461         headers.set(ExtensionHeaderNames.SCHEME.text(), protocol.uriText());
462         headers.set(HttpHeaderNames.USER_AGENT, ARMERIA_USER_AGENT);
463         headers.set(HttpHeaderNames.CONNECTION, HttpHeaderValues.KEEP_ALIVE);
464 
465         ByteBuf contentBuf = request.content();
466         if (contentBuf != null && contentBuf.isReadable()) {
467             headers.set(HttpHeaderNames.CONTENT_LENGTH, contentBuf.readableBytes());
468         }
469 
470         invocation.options().get(ClientOption.HTTP_HEADERS).ifPresent(headers::add);
471         if (ctx.scheme().serializationFormat() != SerializationFormat.NONE) {
472             //we allow a user can set content type and accept headers
473             String mimeType = ctx.scheme().serializationFormat().mimeType();
474             if (!headers.contains(HttpHeaderNames.CONTENT_TYPE)) {
475                 headers.set(HttpHeaderNames.CONTENT_TYPE, mimeType);
476             }
477             if (!headers.contains(HttpHeaderNames.ACCEPT)) {
478                 headers.set(HttpHeaderNames.ACCEPT, mimeType);
479             }
480         }
481 
482         return request;
483     }

485     private static String hostHeader(ServiceInvocationContext ctx) {
486         final int port = ((InetSocketAddress) ctx.remoteAddress()).getPort();
487         return HttpHostHeaderUtil.hostHeader(ctx.host(), port,
488                                              ctx.scheme().sessionProtocol().isTls());
489     }
