1   package com.linecorp.armeria.common.metrics;
2   
3   import java.util.concurrent.TimeUnit;
4   
5   import com.codahale.metrics.Counter;
6   import com.codahale.metrics.Meter;
7   import com.codahale.metrics.Metric;
8   import com.codahale.metrics.Timer;
9   
10  /**
11   * {@link Metric}s for a single request. Used by
12   * {@link com.linecorp.armeria.client.metrics.MetricCollectingClient} and
13   * {@link com.linecorp.armeria.server.metrics.MetricCollectingService} for tracking
14   * request metrics. For internal use only.
15   */
16  public final class DropwizardRequestMetrics {
17  
18      private final String name;
19      private final Timer timer;
20      private final Meter successes;
21      private final Meter failures;
22      private final Counter activeRequests;
23      private final Meter requestBytes;
24      private final Meter responseBytes;
25  
26      public DropwizardRequestMetrics(String name, Timer timer, Meter successes, Meter failures,
27                                      Counter activeRequests,
28                                      Meter requestBytes, Meter responseBytes) {
29          this.name = name;
30          this.timer = timer;
31          this.successes = successes;
32          this.failures = failures;
33          this.activeRequests = activeRequests;
34          this.requestBytes = requestBytes;
35          this.responseBytes = responseBytes;
36      }
37  
38      public Timer.Context time() {
39          return timer.time();
40      }
41  
42      public void updateTime(long durationNanos) {
43          timer.update(durationNanos, TimeUnit.NANOSECONDS);
44      }
45  
46      public void markSuccess() {
47          successes.mark();
48      }
49  
50      public void markFailure() {
51          failures.mark();
52      }
53  
54      public void markStart() {
55          activeRequests.inc();
56      }
57  
58      public void markComplete() {
59          activeRequests.dec();
60      }
61  
62      public void requestBytes(int requestBytes) {
63          this.requestBytes.mark(requestBytes);
64      }
65  
66      public void responseBytes(int responseBytes) {
67          this.responseBytes.mark(responseBytes);
68      }
69  
70      @Override
71      public String toString() {
72          return "<DropwizardRequestMetrics for: " + name + '\n' +
73                 "  requests: " + timer.getCount() + '\n' +
74                 "  successes: " + successes.getCount() + '\n' +
75                 "  failures: " + failures.getCount() + '\n' +
76                 "  activeRequests: " + activeRequests.getCount() + '\n' +
77                 "  requestBytes: " + requestBytes.getCount() + '\n' +
78                 "  responseBytes: " + responseBytes.getCount() + "\n>";
79      }
80  }
