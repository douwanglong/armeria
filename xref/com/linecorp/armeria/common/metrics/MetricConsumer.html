17  package com.linecorp.armeria.common.metrics;
18  
19  import java.util.Objects;
20  import java.util.Optional;
21  
22  import org.slf4j.LoggerFactory;
23  
24  import com.linecorp.armeria.common.Scheme;
25  import com.linecorp.armeria.common.SessionProtocol;
26  
27  public interface MetricConsumer {
28  
29      /**
30       * Invoked when a request is being started
31       *
32       * @param scheme the {@link Scheme} which the invocation has been performed on.
33       */
34      void invocationStarted(Scheme scheme, String hostname, String path, Optional<String> method);
35  
36      /**
37       * Invoked for each request that has been processed
38       *
39       * @param scheme the {@link Scheme} which the invocation has been performed on.
40       * @param code the {@link SessionProtocol}-specific status code that signifies the result of the
41       *             invocation. e.g. HTTP response status code
42       * @param processTimeNanos elapsed nano time processing request
43       * @param requestSize number of bytes in request if possible, otherwise it will be 0
44       * @param responseSize number of bytes in response if possible, otherwise it will be 0
45       * @param started true if invocationStarted() is called before, otherwise false
46       */
47      void invocationComplete(Scheme scheme, int code, long processTimeNanos, int requestSize,
48                              int responseSize, String hostname, String path, Optional<String> method,
49                              boolean started);
50  
51      default MetricConsumer andThen(MetricConsumer other) {
52          Objects.requireNonNull(other, "other");
53          MetricConsumer outer = this;
54          return new MetricConsumer() {
55  
56              @Override
57              public void invocationStarted(Scheme scheme, String hostname, String path, Optional<String> method) {
58                  try {
59                      outer.invocationStarted(scheme, hostname, path, method);
60                  } catch (Throwable e) {
61                      LoggerFactory.getLogger(MetricConsumer.class)
62                                   .warn("invocationStarted() failed with an exception: {}", e);
63                  }
64                  other.invocationStarted(scheme, hostname, path, method);
65              }
66  
67              @Override
68              public void invocationComplete(Scheme scheme, int code, long processTimeNanos, int requestSize,
69                                             int responseSize, String hostname, String path,
70                                             Optional<String> method, boolean started) {
71                  try {
72                      outer.invocationComplete(scheme, code, processTimeNanos, requestSize, responseSize,
73                                               hostname, path, method, started);
74                  } catch (Throwable e) {
75                      LoggerFactory.getLogger(MetricConsumer.class)
76                                   .warn("invocationComplete() failed with an exception: {}", e);
77                  }
78                  other.invocationComplete(scheme, code, processTimeNanos, requestSize, responseSize,
79                                           hostname, path, method, started);
80              }
81          };
82      }
83  }
