17  package com.linecorp.armeria.server.tracing;
18  
19  import com.github.kristofa.brave.Brave;
20  import com.github.kristofa.brave.IdConversion;
21  import com.github.kristofa.brave.SpanId;
22  import com.github.kristofa.brave.TraceData;
23  import com.github.kristofa.brave.http.BraveHttpHeaders;
24  
25  import com.linecorp.armeria.common.ServiceInvocationContext;
26  import com.linecorp.armeria.server.ServiceInvocationHandler;
27  
28  import io.netty.handler.codec.http.HttpHeaders;
29  import io.netty.handler.codec.http.HttpRequest;
30  
31  /**
32   * A {@link TracingServiceInvocationHandler} that uses HTTP headers as a container of trace data.
33   */
34  class HttpTracingServiceInvocationHandler extends TracingServiceInvocationHandler {
35  
36      HttpTracingServiceInvocationHandler(ServiceInvocationHandler handler, Brave brave) {
37          super(handler, brave);
38      }
39  
40      @Override
41      protected TraceData getTraceData(ServiceInvocationContext ctx) {
42          final Object request = ctx.originalRequest();
43          if (request == null || !(request instanceof HttpRequest)) {
44              return TraceData.builder().build();
45          }
46  
47          final HttpHeaders headers = ((HttpRequest) request).headers();
48  
49          // The following HTTP trace header spec is based on
50          // com.github.kristofa.brave.http.HttpServerRequestAdapter#getTraceData
51  
52          final String sampled = headers.get(BraveHttpHeaders.Sampled.getName());
53          if (sampled == null) {
54              // trace data is not specified
55              return TraceData.builder().build();
56          }
57          if ("0".equals(sampled) || "false".equalsIgnoreCase(sampled)) {
58              // this request is not sampled
59              return TraceData.builder().sample(false).build();
60          }
61  
62          final String traceId = headers.get(BraveHttpHeaders.TraceId.getName());
63          final String spanId = headers.get(BraveHttpHeaders.SpanId.getName());
64          if (traceId == null || spanId == null) {
65              // broken trace header
66              return TraceData.builder().build();
67          }
68  
69          // parentSpanId can be null
70          final String parentSpanId = headers.get(BraveHttpHeaders.ParentSpanId.getName());
71  
72          // create new SpanId instance
73          final long traceIdLong = IdConversion.convertToLong(traceId);
74          final long spanIdLong = IdConversion.convertToLong(spanId);
75          final Long parentSpanIdLong = parentSpanId != null ? IdConversion.convertToLong(parentSpanId) : null;
76          final SpanId span = SpanId.create(traceIdLong, spanIdLong, parentSpanIdLong);
77  
78          return TraceData.builder().sample(true).spanId(span).build();
79      }
80  
81  }
